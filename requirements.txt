# Pinned to a single minor version (not just a floor): the text-commit engine
# relies on byte-identical content-stream patching, and PyMuPDF's stream

# let dev (.venv, 1.27.x) and CI/system Python drift apart silently -- see
# docs/PITFALLS.md "PyMuPDF version skew masks runtime-only bugs".
# test_scripts/test_environment_pins.py fails loudly if the resolved version
# leaves this range.
PyMuPDF>=1.27,<1.28

# Image processing for deskew / straighten / optimize / rotated-text preview

# require them and the shipped build bundles them, so they are declared here
# rather than treated as optional. Pillow is floored at 12.2.0 to close the
# image-parser CVEs (CVE-2026-40192 / 42309 / 42310 / 42311, PYSEC-2026-165);
# covered by the CI pip-audit gate.
Pillow>=12.2.0
